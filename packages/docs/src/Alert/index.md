# Alert

基本使用

import { Alert } from '@damw-ui/ui';
import '@damw-ui/ui/dist/mw-ui.css';

export default () => <>
 <Alert  message="message" type="success"></Alert>
 <Alert  message="message" type="error"></Alert>
 <Alert  message="message" type="warning"></Alert>
 <Alert  message="message" type="info"></Alert>
 </>
